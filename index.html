Fade on Death Tutorial:

Code:

local TweenService = game:GetService("TweenService")
local tweenInfo = TweenInfo.new(1)

game.Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function(character)
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid.Died:Connect(function()
                task.wait(0.25)
                for _, part in pairs(character:GetChildren()) do
                    if part:IsA("BasePart") and part.Name ~= "Head" then
                        TweenService:Create(part, tweenInfo, {Transparency = 1}):Play()
                    elseif part:IsA("BasePart") and part.Name == "Head" then
                        TweenService:Create(part, tweenInfo, {Transparency = 1}):Play()
                        TweenService:Create(part:FindFirstChild("face"), tweenInfo, {Transparency = 1}):Play()
                    elseif part:IsA("Accessory") then
                        TweenService:Create(part:FindFirstChild("Handle"), tweenInfo, {Transparency = 1}):Play()
                    end
                end
            end)
        end
    end)
end)
